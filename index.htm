function calcNet(gross, par, strokes) {

const dbl = par + 2 + strokes;

const tri = par + 3 + strokes;

const capped = Math.min(gross, tri);

const tripleCapped = capped < gross;

if (capped > dbl) return {net: dbl, tripleCapped, noCreditCap: true};

return {net: capped - strokes, tripleCapped, noCreditCap: false};

}

 

// Compute strokes per hole from a playing handicap using hole handicap rankings.

// Standard method: allocate one stroke per hole in hdcp order, repeat for each full

// set of 18 until all strokes are distributed.

function computeStrokesFromHcp(playingHcp) {

const strokes = new Array(18).fill(0);

const cap = Math.min(playingHcp, 36); // max 2 strokes per hole

// Sort hole indices by hole handicap (hardest = hdcp 1 gets stroke first)

const byHdcp = Array.from({length:18},(_,i)=>i).sort((a,b)=>COURSE.hdcp[a]-COURSE.hdcp[b]);

for (let s = 0; s < cap; s++) {

strokes[byHdcp[s % 18]]++;

}

return strokes;

}

if (hcp != null) return computeStrokesFromHcp(Math.round(hcp * 0.9)); // 90% playing hcp

return new Array(18).fill(0);

}

// Stableford points for a net score vs par

// Net bogey=1, net par=2, net birdie=4, net eagle+=8, worse=0

function calcStablefordPts(net, par) {

const d = net - par;

if (d >= 1)  return 1;

if (d === 0) return 2;

if (d === -1)return 4;

return 8;

}

// Rank by stableford total (higher = better), requires hole data

function rankByStableford(wk, wkScores) {

const withData = PLAYERS.filter(p => {

const d = wkScores[p];

return d?.holes && Object.keys(d.holes).length > 0;

}).map(p => {

const holes = wkScores[p].holes;

const nine  = wkScores[p].nine || ‘front’;

const st    = getStrokesFor(p);

const idxs  = getHolesForNine(nine);

const sfPts = idxs.reduce((sum, i) => {

if (holes[i] === undefined) return sum;

const { net } = calcNet(holes[i], COURSE.par[i], st[i]);

return sum + calcStablefordPts(net, COURSE.par[i]);

}, 0);

return { player: p, score: sfPts };

}).sort((a,b) => b.score - a.score);

 

if (!withData.length) return [];

const result = [];

let i = 0;

while (i < withData.length) {

let j = i;

while (j < withData.length-1 && withData[j+1].score===withData[i].score) j++;

const pool = FINISH_PTS.slice(i,j+1).reduce((s,v)=>s+v,0);

const pts  = Math.round((pool/(j-i+1))*2)/2;

for (let k=i;k<=j;k++) result.push({player:withData[k].player, score:withData[k].score, rank:i+1, pts, tied:j>i});

i = j+1;

}

return result;

}

}

 

// Calculate 9-hole differential

function calc9Diff(adjGross, side) {

const c = side === ‘Back’ ? COURSE_BACK : COURSE_FRONT;

return Math.round(((adjGross - c.rating) * 113 / c.slope) * 10) / 10;

}

 

// Calculate HI from all diffs18 — rolling avg of last 8

function calcHI(diffs18) {

if (!diffs18.length) return 0;

const window = diffs18.slice(-8);

return Math.round(window.reduce((s,v)=>s+v,0) / window.length * 10) / 10;

}

function addHcptRound() {

const player = document.getElementById(‘hcpt-player-sel’).value;

const side = document.getElementById(‘hcpt-side’).value;

const adjGross = Number(document.getElementById(‘hcpt-adjgross’).value);

if (!player || !adjGross) { toast(‘⚠️ Enter adj gross score’); return; }

const data = getHcpRounds(player);

const rounds = data.rounds || [];

const hiBefore = rounds.length ? rounds[rounds.length-1].hiAfter : data.startHI;

const diff9 = calc9Diff(adjGross, side);

const exp9 = Math.round(hiBefore / 2 * 10) / 10;

const diff18 = Math.round((diff9 + exp9) * 10) / 10;

const allDiff18 = rounds.map(r=>r.diff18).concat([diff18]);

const hiAfter = calcHI(allDiff18);

rounds.push({ rnd: rounds.length + 1, side, adjGross, diff9, hiBefore, exp9, diff18, hiAfter });

data.rounds = rounds;

saveHcpRounds(player, data);

 

// Also update the handicap in the main hcaps store so strokes per hole update

const hcaps = getHandicaps();

hcaps[player] = hiAfter;

saveHandicaps_(hcaps);

 

renderHcpTracker();

toast(`✅ Round added · ${player} HI: ${hiBefore} → ${hiAfter}`);

}

 

function deleteHcptRound(player, idx) {

if (!confirm(‘Delete this round? HI will be recalculated.’)) return;

const data = getHcpRounds(player);

data.rounds.splice(idx, 1);

// Recalculate all hiAfter values

data.rounds.forEach((r, i) => {

const hiBefore = i === 0 ? data.startHI : data.rounds[i-1].hiAfter;

const allDiff18 = data.rounds.slice(0, i+1).map(x=>x.diff18);

r.rnd = i + 1;

r.hiBefore = hiBefore;

r.exp9 = Math.round(hiBefore / 2 * 10) / 10;

r.diff18 = Math.round((r.diff9 + r.exp9) * 10) / 10;

r.hiAfter = calcHI(allDiff18);

});

saveHcpRounds(player, data);

// Update main hcap

const hcaps = getHandicaps();

const lastRound = data.rounds[data.rounds.length-1];

if (lastRound) hcaps[player] = lastRound.hiAfter;

saveHandicaps_(hcaps);

renderHcpTracker();

toast(‘Round deleted · HI recalculated’);

}

const COURSE_FRONT = {rating:33.5, slope:117, par:35};

const COURSE_BACK  = {rating:33.7, slope:118, par:35};

 

function hiToCourseHcp(hi, side) {

const c = side===‘back’ ? COURSE_BACK : COURSE_FRONT;

return Math.round(hi * (c.slope/113) + (c.rating - c.par));

}

function courseHcpToPlaying(courseHcp) {

return Math.round(courseHcp * 0.9);

}

function onHIInput(pid) {

const inp = document.getElementById(‘hi_’+pid);

const val = inp.value.trim();

if (!val || isNaN(Number(val))) {

[‘fc’,‘fp’,‘bc’,‘bp’].forEach(k => { const e=document.getElementById(`hi_${k}_${pid}`); if(e) e.textContent=’—’; });

return;

}

const hi = Number(val);

const fc = hiToCourseHcp(hi,‘front’), fp = courseHcpToPlaying(fc);

const bc = hiToCourseHcp(hi,‘back’),  bp = courseHcpToPlaying(bc);

document.getElementById(`hi_fc_${pid}`).textContent = fc;

document.getElementById(`hi_fp_${pid}`).textContent = fp;

document.getElementById(`hi_bc_${pid}`).textContent = bc;

document.getElementById(`hi_bp_${pid}`).textContent = bp;

}

 

function saveHIs() {

if (!requireAuth()) return;

PLAYERS = getPlayers();

const hcaps = getHandicaps();

let updated = 0;

PLAYERS.forEach(p => {

const inp = document.getElementById(‘hi_’+p.replace(/\W/g,’*’));

if (inp && inp.value.trim() !== ‘’) {

hcaps[p] = Number(inp.value);

updated++;

}

});

saveHandicaps*(hcaps);

renderHISummary();

toast(`✅ ${updated} HIs saved · Strokes per hole updated`);

}
